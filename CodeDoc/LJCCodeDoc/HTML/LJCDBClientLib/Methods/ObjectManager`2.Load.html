// Loads multiple Person objects.
private static Persons Load(ObjectManager<Person, Persons> personManager)
{
  Persons retValue;

  // Create Key Columns.
  var keyColumns = new DBColumns()
  {

  };

  // Load the records.
  retValue = personManager.Load(keyColumns);
  return retValue;
